using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// Must be on root of vehicle.
/// Automatically finds all wheel colliders and trail renderers on the vehicle and binds them together.
/// 
/// Made by StCost, for COLLAPSE MACHINE game project.
/// See flaws in blog post: https://dreamingsaints.github.io/blog/51-wheelstrailsrenderer/
/// </summary>
[DisallowMultipleComponent]
public class WheelsTrailsRenderer : MonoBehaviour
{
    [System.Serializable]
    private struct WheelTrail
    {
        public WheelCollider wheelCollider;
        public TrailRenderer trailRenderer;
    }

    [SerializeField] private List<WheelTrail> wheelTrails = new();
    private readonly static Vector3 TRAIL_OFFSET = new(0, 0.02f, 0); // Place trail slightly above the ground hit point, to avoid z-fighting.

#if UNITY_EDITOR
    private void Reset()
    {
        wheelTrails.Clear();

        foreach (WheelCollider wheelCollider in GetComponentsInChildren<WheelCollider>())
        {
            TrailRenderer trailRenderer = wheelCollider.GetComponentInChildren<TrailRenderer>();

            // skip if wheel has no trail renderer inside
            if (!trailRenderer)
                continue;

            // validate trail renderer settings
            {
                bool changed = false;
                // must be transform-Z aligned with the wheel collider, so can look up relative to ground
                if (trailRenderer.alignment != LineAlignment.TransformZ)
                {
                    trailRenderer.alignment = LineAlignment.TransformZ;
                    changed = true;
                }

                // texture mode must be "Repeat Per Segment" to correctly tile along the wheel trail
                if (trailRenderer.textureMode != LineTextureMode.RepeatPerSegment)
                {
                    trailRenderer.textureMode = LineTextureMode.RepeatPerSegment;
                    changed = true;
                }

                // no shadows needed
                var off = UnityEngine.Rendering.ShadowCastingMode.Off;
                if (trailRenderer.shadowCastingMode != off)
                {
                    trailRenderer.shadowCastingMode = off;
                    changed = true;
                }

                // generate light data to properly lit trail. Anyway trail can't be unlit shader, otherwise ugly
                if (!trailRenderer.generateLightingData)
                {
                    trailRenderer.generateLightingData = true;
                    changed = true;
                }

                // disable unused active color space
                if (trailRenderer.applyActiveColorSpace)
                {
                    trailRenderer.applyActiveColorSpace = false;
                    changed = true;
                }

                if (changed)
                {
                    UnityEditor.EditorUtility.SetDirty(trailRenderer);
                    Debug.Log($"[VehicleWheelTrail] Trail renderer '{trailRenderer.name}' was changed to behave better with wheel trails", trailRenderer);
                }
            }

            wheelTrails.Add(new WheelTrail { wheelCollider = wheelCollider, trailRenderer = trailRenderer });
        }
    }
#endif

    private void Update()
    {
        foreach (WheelTrail wheelTrail in wheelTrails)
            UpdateWheelTrail(wheelTrail);
    }

    private static void UpdateWheelTrail(WheelTrail wheelTrail)
    {
        var isGrounded = wheelTrail.wheelCollider.GetGroundHit(out var wheelHit);

        wheelTrail.trailRenderer.emitting = isGrounded;

        if (isGrounded)
            wheelTrail.trailRenderer.transform.SetPositionAndRotation(
                wheelHit.point + TRAIL_OFFSET,
                Quaternion.LookRotation(-wheelHit.normal, Vector3.up)
            );
    }
}

<details>
<summary>Pros</summary>

- Single basic script on root of vehicle
- Leaves trails while driving. If jumped - gap in trail.
- Any lit transparent shader or material
    
</details>

<details>
<summary>Cons</summary>

- Start and end of trail has triangle merging 2 points into 1. Instead of desired straights cut
- After jump - end or start of trail slightly points upwards from ground. Making visible triangle barely detach from ground
